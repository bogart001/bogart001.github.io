# 2、循环实现数组 map 方法

const selfMap = function (fn, context) {
  let arr = Array.prototype.slice.call(this);
  let mappedArr = Array();
  for (let i = 0; i < arr.length; i++) {
    if (!arr.hasOwnProperty(i)) continue;
    mappedArr[i] = fn.call(context, arr[i], i, this);
  }
};
